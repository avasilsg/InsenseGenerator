
type sensorRading is struct (
	integer solar;
	integer photo;
	real temp;
	real humid;)

type ISensorReading is interface
(
	in bool ticks;
	out sensorReading printerChan ;
	out sensorReading ledsChan;
)
component SensorReader presents ISensorReader
{


	constructor ()
	{
	}

	behaviour 
	{
		receive tick from ticks
		send reading on ledsChan
		send reading on printerChan
	}
}
type IPrintOutput is interface
(
	in sensorReading input;
)
component PrintOutput presents IPrintOutput
{

	cycle = $0
	constructor ()
	{
	}

	behaviour 
	{

		printInt(cycle)

		printReal(reading.photo)
	}
}
type ILedOutput is interface
(
	in sensorReading input;
)
component LedOutput presents ILedOutput
{

	avgTemp = 0.0
	avgSolar = 0.0
	avgSolar = 0.0
	constructor ()
	{
	}

	behaviour 
	{
		receive reading from input
	}
}
connect sensorReader.printerChan to printOutput.input
